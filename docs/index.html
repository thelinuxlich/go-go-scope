Quick Start

import { scope, exponentialBackoff } from 'go-go-scope'

// Build a resilient API in 30 lines
await using s = scope({
  timeout: 10000,
  circuitBreaker: { failureThreshold: 3, resetTimeout: 30000 }
})

// Race multiple data sources - first to succeed wins
const [err, data] = await s.race([
  () => fetchFromPrimary(),
  () => fetchFromCache(),
  () => fetchFromBackup()
])

// Parallel processing with retry and backoff
const [pErr, results] = await s.parallel(
  ids.map(id => () => 
    s.task(() => processItem(id), {
      retry: {
        maxRetries: 3,
        delay: exponentialBackoff({ initial: 100, max: 2000 })
      }
    })
  ),
  { concurrency: 5 }
)

// Everything cancels and cleans up automatically
// No memory leaks, no hanging promises, no manual cleanup

The Scope

A Scope is a container for async operations. Think of it as a "workspace" for your tasks with automatic cleanup and cancellation propagation.

Key Features

  • Contains tasks - All tasks are created within a scope
  • Propagates cancellation - When the scope ends, tasks are cancelled
  • Manages resources - Services you provide are cleaned up automatically
  • Tracks execution - Knows what tasks are running
  • Collects telemetry - Creates spans for tracing

Creating a Scope

// Simple scope
await using s = scope()

// With timeout
await using s = scope({ timeout: 5000 })

// With OpenTelemetry tracing
await using s = scope({
  name: 'my-operation',
  tracer: trace.getTracer('my-app')
})

// With concurrency limit
await using s = scope({ concurrency: 3 })

// With circuit breaker
await using s = scope({
  circuitBreaker: {
    failureThreshold: 5,
    resetTimeout: 30000
  }
})

The Task

A Task is a unit of work within a scope. It's like a Promise, but:

  • Lazy - Doesn't start until you await it
  • Cancellable - Respects the AbortSignal
  • Returns a Result - Never throws, always returns [error, value]

Lazy Execution

await using s = scope()

// Create tasks (doesn't start yet!)
const primary = s.task(() => fetchFromPrimary())
const fallback = s.task(() => fetchFromFallback())

// Try primary first
const [err, result] = await primary
if (err) {
  // Only now does fallback start!
  const [fallbackErr, fallbackResult] = await fallback
  return fallbackResult
}
return result

Task Context

Task functions receive a context object with useful properties:

await s.task(async ({ services, signal, logger, context }) => {
  // services: Your provided services (DI)
  // signal: AbortSignal for cancellation
  // logger: Structured logger (debug, info, warn, error)
  // context: Request-scoped context object from scope
  
  logger.info('Starting operation', { requestId: context.requestId })
  
  const data = await fetchData({ signal })
  return data
})

Channels

Channels provide Go-style concurrent communication with configurable backpressure strategies.

Basic Usage

await using s = scope()
const ch = s.channel<string>(100)  // Buffer capacity of 100

// Producer
s.task(async () => {
  for (const log of logs) {
    await ch.send(log)  // Blocks if buffer full (default)
  }
  ch.close()
})

// Consumer with native async iteration
for await (const log of ch) {
  await processLog(log)
}

Priority Channels

const pq = s.priorityChannel<{ value: Task; priority: number }>({
  capacity: 100,
  comparator: (a, b) => a.priority - b.priority
})

// Send tasks with priorities
await pq.send({ value: task1, priority: 3 })
await pq.send({ value: task2, priority: 1 })  // Higher priority

// Receive in priority order
const item = await pq.receive() // task2 first

import {

// Race multiple signals

Examples

Real-world patterns showcasing the power of structured concurrency.

🔥 Resilient HTTP Client with Retry & Circuit Breaker

Fetch data with exponential backoff, circuit breaker protection, and graceful fallback - all automatically cleaned up.

import { scope, exponentialBackoff } from 'go-go-scope'

async function fetchUserWithResilience(userId: string) {
  await using s = scope({
    timeout: 30000,
    circuitBreaker: {
      failureThreshold: 5,
      resetTimeout: 30000,
      onStateChange: (from, to) => console.log(`Circuit: ${from} → ${to}`)
    }
  })

  // Try primary API with aggressive retry
  const primary = s.task(
    () => fetch(`https://api.example.com/users/${userId}`),
    {
      retry: {
        maxRetries: 5,
        delay: exponentialBackoff({ initial: 100, max: 5000 }),
        retryCondition: (err) => err.status >= 500
      }
    }
  )

  // Fallback to cache if API fails
  const fallback = s.task(() => redisGet(`user:${userId}`))

  // Race them - if primary fails quickly, use cache
  const [err, user] = await s.race([
    () => primary,
    () => fallback
  ], { requireSuccess: true })

  return err ? null : user
  // Everything cancelled and cleaned up automatically!
}

📊 Parallel Data Processing with Stream Pipeline

Process millions of records with backpressure, batching, and real-time metrics - production-ready data pipeline.

import { scope } from 'go-go-scope'
import { streamPlugin } from '@go-go-scope/stream'
import { metricsPlugin } from '@go-go-scope/metrics'

async function processDataPipeline(source: AsyncIterable<RawEvent>) {
  await using s = scope({
    name: 'data-pipeline',
    plugins: [streamPlugin(), metricsPlugin()],
    concurrency: 10
  })

  const processed = s.counter('events_processed')
  const latency = s.histogram('processing_latency_ms')

  const [err, results] = await s.stream(source)
    .map(async (event) => {
      const start = performance.now()
      const enriched = await enrichEvent(event)
      latency.record(performance.now() - start)
      return enriched
    })
    .filter(e => e.isValid)
    .buffer(100)           // Batch for efficient DB writes
    .tap((batch) => {
      processed.inc(batch.length)
      console.log(`Processed ${batch.length} events`)
    })
    .flatMap(async (batch) => {
      // Parallel database inserts with concurrency limit
      const [, results] = await s.parallel(
        batch.map(e => () => dbInsert(e)),
        { concurrency: 5 }
      )
      return results
    })
    .take(10000)          // Limit to 10k events
    .toArray()            // Collect results

  if (err) console.error('Pipeline failed:', err)
  else console.log(`Completed: ${processed.value()} events`)
  
  return results
}

🔄 Real-time WebSocket Fan-out with Channels

Broadcast real-time updates to thousands of clients with automatic cleanup when connections drop.

import { scope } from 'go-go-scope'

async function runWebSocketServer(wss: WebSocketServer) {
  await using serverScope = scope({ name: 'ws-server' })

  // Broadcast channel for price updates
  const prices = serverScope.broadcast<PriceUpdate>()

  // Poll external API for price changes
  const poller = serverScope.poll(
    () => fetchPrices(),
    (update) => prices.send(update),
    { interval: 1000 }
  )

  // Handle client connections
  wss.on('connection', (ws, req) => {
    // Each client gets its own scope
    serverScope.task(async () => {
      await using clientScope = scope({
        parent: serverScope,
        name: `client-${req.socket.remotePort}`
      })

      // Subscribe to relevant symbols
      const symbols = new Set<string>()
      
      await prices.subscribe((update) => {
        if (symbols.has(update.symbol)) {
          ws.send(JSON.stringify(update))
        }
      })

      // Handle client messages
      ws.on('message', (data) => {
        const msg = JSON.parse(data.toString())
        if (msg.action === 'subscribe') {
          symbols.add(msg.symbol)
        }
      })

      // Cleanup on disconnect - automatic!
      await new Promise((resolve) => ws.on('close', resolve))
    })
  })

  // Server runs until SIGTERM
  await waitForShutdown()
  // All clients disconnected, polling stopped, resources freed
}

🗄️ Database Transaction with Connection Pooling

Complex multi-table transaction with connection pooling, automatic rollback, and deadlock detection.

import { scope } from 'go-go-scope'
import { deadlockDetectorPlugin } from '@go-go-scope/deadlock-detector'

async function transferFunds(
  fromId: string,
  toId: string,
  amount: number
) {
  await using s = scope({
    name: 'transfer',
    timeout: 10000,
    plugins: [deadlockDetectorPlugin({ timeout: 5000 })]
  })

  // Provide database pool as a service
  await s.provide('db', 
    () => createConnection(),
    (conn) => conn.end()
  )

  // Create connection pool
  const pool = s.pool({
    factory: () => s.use('db'),
    min: 2, max: 10,
    acquireTimeout: 5000
  })

  return pool.execute(async (conn) => {
    await conn.query('BEGIN')

    // Deduct from sender
    const [err1] = await s.task(() =>
      conn.query('UPDATE accounts SET balance = balance - $1 WHERE id = $2',
        [amount, fromId]
      )
    )
    if (err1) {
      await conn.query('ROLLBACK')
      return { success: false, error: err1 }
    }

    // Add to receiver
    const [err2] = await s.task(() =>
      conn.query('UPDATE accounts SET balance = balance + $1 WHERE id = $2',
        [amount, toId]
      )
    )
    if (err2) {
      await conn.query('ROLLBACK')
      return { success: false, error: err2 }
    }

    // Record transaction
    const [err3] = await s.task(() =>
      conn.query(
        'INSERT INTO transactions (from_id, to_id, amount) VALUES ($1, $2, $3)',
        [fromId, toId, amount]
      )
    )
    if (err3) {
      await conn.query('ROLLBACK')
      return { success: false, error: err3 }
    }

    await conn.query('COMMIT')
    return { success: true }
  })
  // Pool drained, connections closed automatically
}

🚀 Distributed Job Scheduler with Persistence

Production-ready cron job scheduler with distributed locking, idempotency, and circuit breaker.

import { Scheduler } from '@go-go-scope/scheduler'
import { RedisIdempotencyAdapter } from '@go-go-scope/persistence-redis'

async function setupScheduledJobs() {
  const scheduler = new Scheduler({
    storage: await createPostgresStorage(),
    checkInterval: 5000,  // Check every 5 seconds
    persistence: {
      idempotency: new RedisIdempotencyAdapter(redis)
    }
  })

  // Daily report generation
  await scheduler.createSchedule({
    name: 'daily-report',
    cron: '0 6 * * *',  // 6 AM daily
    timezone: 'America/New_York',
    handler: async (job, { scope }) => {
      // Each job runs in its own scope
      await using s = scope

      const [err, report] = await s.task(
        () => generateReport(job.payload.date),
        {
          // Idempotent - won't run twice for same date
          idempotency: { key: `report:${job.payload.date}`, ttl: 86400000 },
          retry: 'exponential',
          timeout: 300000  // 5 minutes
        }
      )

      if (err) {
        console.error('Report failed:', err)
        throw err  // Will retry based on schedule config
      }

      await s.task(() => emailReport(report))
    },
    options: {
      maxRetries: 3,
      retryDelay: 60000,  // 1 minute between retries
      timeout: 600000        // 10 minute total timeout
    }
  })

  // Health check endpoint
  scheduler.on('jobFailed', ({ job, error, permanent }) => {
    if (permanent) {
      alertPagerDuty(`Job ${job.scheduleName} permanently failed: ${error}`)
    }
  })

  await scheduler.start()
  console.log('Scheduler running')
}

🔒 Distributed Lock for Critical Operations

Prevent race conditions in multi-instance deployments with Redis-backed distributed locks.

import { scope } from 'go-go-scope'
import { RedisLockProvider } from '@go-go-scope/persistence-redis'

async function processInventoryUpdate(
  productId: string,
  quantity: number
) {
  await using s = scope({
    name: 'inventory-update',
    timeout: 30000,
    persistence: {
      lock: new RedisLockProvider(redis)
    }
  })

  // Acquire distributed lock - only one instance can process this product
  const lockKey = `inventory:${productId}`
  const lock = await s.acquireLock(lockKey, {
    ttl: 10000,        // Lock expires after 10s
    retry: 5,          // Retry 5 times
    retryDelay: 500   // Wait 500ms between retries
  })

  if (!lock) {
    return { success: false, error: 'Could not acquire lock - another instance is processing' }
  }

  // Lock acquired - safely check and update inventory
  const [err, currentStock] = await s.task(
    () => getCurrentStock(productId)
  )

  if (err) {
    await lock.release()
    return { success: false, error: err }
  }

  if (currentStock < quantity) {
    await lock.release()
    return { success: false, error: 'Insufficient stock' }
  }

  // Update inventory within the lock
  const [updateErr] = await s.task(() =>
    updateStock(productId, currentStock - quantity)
  )

  // Extend lock if processing takes longer than expected
  await lock.extend(5000)

  // Publish stock change event
  await s.task(() =>
    publishEvent('stock.updated', { productId, newStock: currentStock - quantity })
  )

  // Release lock when done
  await lock.release()

  return { success: !updateErr, error: updateErr }
  // Lock automatically released when scope exits (even on error!)
}

Core Types

Result<E, T>

Result tuple for type-safe error handling. Tasks always return [error, value] tuples - they never throw.

type Result<E, T> = readonly [E | undefined, T | undefined]
type Success<T> = readonly [undefined, T]
type Failure<E> = readonly [E, undefined]

ScopeOptions<TServices>

Configuration options for creating a scope.

interface ScopeOptions<TServices extends Record<string, unknown>> {
  /** Timeout in milliseconds - scope aborts after this duration */
  timeout?: number
  
  /** Parent AbortSignal to link cancellation */
  signal?: AbortSignal
  
  /** Scope name for tracing and debugging */
  name?: string
  
  /** Maximum concurrent tasks (creates Semaphore) */
  concurrency?: number
  
  /** Circuit breaker configuration for all tasks */
  circuitBreaker?: CircuitBreakerOptions
  
  /** Parent scope to inherit signal, services, and options */
  parent?: Scope<TServices>
  
  /** Lifecycle hooks for scope events */
  hooks?: ScopeHooks
  
  /** Custom logger for structured logging */
  logger?: Logger
  
  /** Minimum log level (debug, info, warn, error) */
  logLevel?: "debug" | "info" | "warn" | "error"
  
  /** Persistence providers for distributed features */
  persistence?: PersistenceProviders
  
  /** Default TTL for idempotency keys (ms) */
  idempotency?: { defaultTTL?: number }
  
  /** Task pooling configuration for reduced GC */
  taskPooling?: { enabled?: boolean; maxSize?: number }
  
  /** Request-scoped context accessible in tasks */
  context?: Record<string, unknown>
  
  /** Plugins to extend scope functionality */
  plugins?: ScopePlugin[]
}

TaskOptions<E>

Options for spawning a task. errorClass and systemErrorClass are mutually exclusive.

interface TaskOptionsBase {
  /** OpenTelemetry tracing options */
  otel?: {
    name?: string
    attributes?: Record<string, unknown>
  }
  
  /** Retry configuration */
  retry?: {
    maxRetries?: number      // Default: 3
    delay?: number | RetryDelayFn  // Delay between retries
    retryCondition?: (error: unknown) => boolean
    onRetry?: (error: unknown, attempt: number) => void
  } | "exponential" | "linear" | "fixed"
  
  /** Task timeout in milliseconds */
  timeout?: number
  
  /** Cleanup function after task completes */
  onCleanup?: () => void | Promise<void>
  
  /** Deduplication key - in-flight tasks share results */
  dedupe?: string | symbol
  
  /** Memoization config - caches successful results */
  memo?: { key: string | symbol; ttl: number }
  
  /** Error context for debugging */
  errorContext?: Record<string, unknown>
  
  /** Idempotency config (requires persistence provider) */
  idempotency?: {
    key: string | ((...args: unknown[]) => string)
    ttl?: number
  }
}

/** Wraps ALL errors in the provided class */
interface TaskOptionsWithErrorClass<E> extends TaskOptionsBase {
  errorClass: ErrorConstructor<E>
  systemErrorClass?: never
}

/** Only wraps untagged (system) errors */
interface TaskOptionsWithSystemErrorClass<E> extends TaskOptionsBase {
  errorClass?: never
  systemErrorClass: ErrorConstructor<E>
}

type TaskOptions<E> = TaskOptionsWithErrorClass<E> |

CircuitBreakerOptions

Configuration for circuit breaker fault tolerance.

interface CircuitBreakerOptions {
  /** Failures before opening circuit. Default: 5 */
  failureThreshold?: number
  
  /** Time before attempting reset (ms). Default: 30000 */
  resetTimeout?: number
  
  /** Consecutive successes to close from half-open. Default: 1 */
  successThreshold?: number
  
  /** Called on state change */
  onStateChange?: (
    from: CircuitState,
    to: CircuitState,
    failureCount: number
  ) => void
  
  /** Called when circuit opens */
  onOpen?: (failureCount: number) => void
  
  /** Called when circuit closes */
  onClose?: () => void
  
  /** Called when entering half-open */
  onHalfOpen?: () => void
  
  /** Advanced configuration */
  advanced?: {
    /** Enable adaptive threshold based on error rate */
    adaptiveThreshold?: boolean
    minThreshold?: number           // Default: 2
    maxThreshold?: number           // Default: 10
    errorRateWindowMs?: number      // Default: 60000
    onThresholdAdapt?: (newThreshold: number, errorRate: number) => void
    
    /** Use sliding window for failure counting */
    slidingWindow?: boolean
    slidingWindowSizeMs?: number  // Default: 60000
  }
}

type CircuitState = "closed" | "open" | "half-open"

ChannelOptions<T>

interface ChannelOptions<T> {
  /** Buffer capacity. Default: 0 (unbuffered) */
  capacity?: number
  
  /** Backpressure strategy. Default: 'block' */
  backpressure?: BackpressureStrategy
  
  /** Called when value is dropped due to backpressure */
  onDrop?: (value: T) => void
  
  /** Time window for 'sample' strategy (ms). Default: 1000 */

type BackpressureStrategy = 
  | "block"        // Wait until space available (default)
  | "drop-oldest"  // Remove oldest item to make room
  | "drop-latest"  // Drop new item when full
  | "error"        // Throw ChannelFullError when full
  | "sample"       // Keep only values within time window

    () => fetchUser(ctx.params.id)
  )
  
  ctx.body = err ? { error: err.message } : user
})
